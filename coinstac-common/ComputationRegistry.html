Example
const instance = new ComputationRegistry({
  github: new GitHubApi({...}),
  path: path.join(__dirname, 'computations'),
  registry: [{
    name: 'my-computation',
    tags: ['1.0.0'],
    url: 'https://github.com/MRN-Code/coinstac-my-computation',
  }, {
    name: 'my-other-computation',

});
